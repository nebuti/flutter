185    bool isOperator(String x) {
186      if (x == '/' || x == 'x' || x == '-' || x == '+' || x == '=') {
187        return true;
188      }
189      return false;
190    }

192  // function to calculate the input operation
193    void equalPressed() {
194      String finaluserinput = userInput;
195      finaluserinput = userInput.replaceAll('x', '*');
196  
197      Parser p = Parser();
198      Expression exp = p.parse(finaluserinput);
199      ContextModel cm = ContextModel();
200      double eval = exp.evaluate(EvaluationType.REAL, cm);
201      answer = eval.toString();
202    }
